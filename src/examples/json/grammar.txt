json   = _ value _
value  =  Str / Arr / Obj / num / lit
Obj    = '{'_ (memb (_','_ memb)*)? _'}'
memb   = Str _':'_ value
Arr    = '['_ (value (_','_ value)*)? _']'
Str    = '"' chars* '"'
chars  = ~[\u0000-\u001F"\]+ / '\' esc
esc    = ["\/bfnrt] / 'u' [0-9a-fA-F]*4
num    = _int _frac? _exp?
_int   = '-'? ([1-9] [0-9]* / '0')
_frac  = '.' [0-9]+
_exp   = [eE] [+-]? [0-9]+
lit    = 'true' / 'false' / 'null'
_      = [ \t\n\r]*
